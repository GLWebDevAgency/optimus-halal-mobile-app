Create `web/src/components/animations/smooth-scroll.tsx`:

```tsx
"use client";

import { useEffect, useRef } from "react";
import Lenis from "lenis";

export function SmoothScroll({ children }: { children: React.ReactNode }) {
  const lenisRef = useRef<Lenis | null>(null);

  useEffect(() => {
    // Respect prefers-reduced-motion
    const prefersReduced = window.matchMedia(
      "(prefers-reduced-motion: reduce)"
    ).matches;
    if (prefersReduced) return;

    const lenis = new Lenis({
      lerp: 0.08,
      smoothWheel: true,
    });
    lenisRef.current = lenis;

    let rafId: number;
    function raf(time: number) {
      lenis.raf(time);
      rafId = requestAnimationFrame(raf);
    }
    rafId = requestAnimationFrame(raf);

    return () => {
      cancelAnimationFrame(rafId);
      lenis.destroy();
      lenisRef.current = null;
    };
  }, []);

  return <>{children}</>;
}
```

Create `web/src/components/animations/split-text.tsx`:

```tsx
"use client";

import { useRef } from "react";
import { motion, useInView } from "motion/react";

interface SplitTextProps {
  children: string;
  className?: string;
  as?: "h1" | "h2" | "h3" | "p" | "span";
  delay?: number;
  stagger?: number;
  once?: boolean;
}

export function SplitText({
  children,
  className,
  as: Tag = "h2",
  delay = 0,
  stagger = 0.02,
  once = true,
}: SplitTextProps) {
  const ref = useRef<HTMLElement>(null);
  const isInView = useInView(ref, { once, amount: 0.4 });

  // Check prefers-reduced-motion via CSS — fallback to simple fade
  const chars = children.split("");

  return (
    <Tag ref={ref as any} className={className} aria-label={children}>
      <span className="sr-only">{children}</span>
      <span aria-hidden="true" style={{ perspective: "800px" }}>
        {chars.map((char, i) => (
          <motion.span
            key={`${char}-${i}`}
            style={{
              display: "inline-block",
              whiteSpace: char === " " ? "pre" : undefined,
            }}
            initial={{ opacity: 0, rotateX: 90, y: 20 }}
            animate={
              isInView
                ? { opacity: 1, rotateX: 0, y: 0 }
                : { opacity: 0, rotateX: 90, y: 20 }
            }
            transition={{
              duration: 0.5,
              delay: delay + i * stagger,
              ease: [0.25, 0.1, 0.25, 1],
            }}
          >
            {char}
          </motion.span>
        ))}
      </span>
    </Tag>
  );
}
```

- [ ] **Step 2: Create AnimateIn component**

Create `web/src/components/animations/animate-in.tsx`:

```tsx
"use client";

import { useRef } from "react";
import { motion, useInView, type Variants } from "motion/react";

type AnimateInVariant = "fadeUp" | "fadeDown" | "fadeLeft" | "fadeRight" | "scaleIn" | "blur";

interface AnimateInProps {
  children: React.ReactNode;
  className?: string;
  variant?: AnimateInVariant;
  delay?: number;
  duration?: number;
  once?: boolean;
  amount?: number;
}

const variants: Record<AnimateInVariant, Variants> = {
  fadeUp: {
    hidden: { opacity: 0, y: 24 },
    visible: { opacity: 1, y: 0 },
  },
  fadeDown: {
    hidden: { opacity: 0, y: -24 },
    visible: { opacity: 1, y: 0 },
  },
  fadeLeft: {
    hidden: { opacity: 0, x: -24 },
    visible: { opacity: 1, x: 0 },
  },
  fadeRight: {
    hidden: { opacity: 0, x: 24 },
    visible: { opacity: 1, x: 0 },
  },
  scaleIn: {
    hidden: { opacity: 0, scale: 0.8 },
    visible: { opacity: 1, scale: 1 },
  },
  blur: {
    hidden: { opacity: 0, filter: "blur(6px)" },
    visible: { opacity: 1, filter: "blur(0px)" },
  },
};

export function AnimateIn({
  children,
  className,
  variant = "fadeUp",
  delay = 0,
  duration = 0.6,
  once = true,
  amount = 0.4,
}: AnimateInProps) {
  const ref = useRef<HTMLDivElement>(null);
  const isInView = useInView(ref, { once, amount });

  return (
    <motion.div
      ref={ref}
      className={className}
      initial="hidden"
      animate={isInView ? "visible" : "hidden"}
      variants={variants[variant]}
      transition={{
        duration,
        delay,
        ease: [0.25, 0.1, 0.25, 1],
      }}
    >
      {children}
    </motion.div>
  );
}

/** Stagger container — wrap multiple AnimateIn children */
export function Stagger({
  children,
  className,
  stagger = 0.05,
  once = true,
}: {
  children: React.ReactNode;
  className?: string;
  stagger?: number;
  once?: boolean;
}) {
  const ref = useRef<HTMLDivElement>(null);
  const isInView = useInView(ref, { once, amount: 0.3 });

  return (
    <motion.div
      ref={ref}
      className={className}
      initial="hidden"
      animate={isInView ? "visible" : "hidden"}
      variants={{
        hidden: {},
        visible: { transition: { staggerChildren: stagger } },
      }}
    >
      {children}
    </motion.div>
  );
}

/** Stagger child — use inside <Stagger> */
export function StaggerItem({
  children,
  className,
  variant = "fadeUp",
}: {
  children: React.ReactNode;
  className?: string;
  variant?: AnimateInVariant;
}) {
  return (
    <motion.div
      className={className}
      variants={variants[variant]}
      transition={{ duration: 0.5, ease: [0.25, 0.1, 0.25, 1] }}
    >
      {children}
    </motion.div>
  );
}
```

- [ ] **Step 2: Create PhoneScreenManager**

Create `web/src/components/phone/phone-screen-manager.tsx`:

Manages which screen is displayed and animates transitions. Receives `activeScreen` (string key) and a map of screen components. Determines transition direction based on screen metadata.

```tsx
"use client";

import { AnimatePresence, motion } from "motion/react";
import { useMemo } from "react";

export interface ScreenConfig {
  key: string;
  component: React.ReactNode;
  category: string; // Same category → horizontal, different → vertical
}

interface PhoneScreenManagerProps {
  screens: ScreenConfig[];
  activeScreen: string;
}

export function PhoneScreenManager({
  screens,
  activeScreen,
}: PhoneScreenManagerProps) {
  const screenMap = useMemo(
    () => new Map(screens.map((s) => [s.key, s])),
    [screens]
  );

  const current = screenMap.get(activeScreen);
  const currentIndex = screens.findIndex((s) => s.key === activeScreen);

  // Determine transition direction
  const getDirection = (enteringKey: string) => {
    const entering = screenMap.get(enteringKey);
    const prevIndex = currentIndex > 0 ? currentIndex - 1 : 0;
    const prev = screens[prevIndex];
    if (!entering || !prev) return "vertical";
    return entering.category === prev.category ? "horizontal" : "vertical";
  };

  const direction = getDirection(activeScreen);
  const isHorizontal = direction === "horizontal";

  return (
    <div className="relative h-full w-full overflow-hidden bg-[#0f0f0f]">
      <AnimatePresence mode="wait">
        <motion.div
          key={activeScreen}
          className="absolute inset-0"
          initial={{
            x: isHorizontal ? "100%" : 0,
            y: isHorizontal ? 0 : "100%",
            opacity: 0.5,
          }}
          animate={{ x: 0, y: 0, opacity: 1 }}
          exit={{
            x: isHorizontal ? "-30%" : 0,
            y: isHorizontal ? 0 : "-30%",
            opacity: 0,
          }}
          transition={{
            type: "spring",
            stiffness: 100,
            damping: 20,
            mass: 0.8,
          }}
        >
          {current?.component}
        </motion.div>
      </AnimatePresence>
    </div>
  );
}
```

## Task 7: StickyPhone Container

**Files:**

- [ ] **Step 1: Create StickyPhone**

Create `web/src/components/phone/sticky-phone.tsx`:

Sticky container for the phone on the right side of the two-column grid. Uses `position: sticky; top: 50%; transform: translateY(-50%)` to vertically center in the viewport.

Scroll-driven micro-movements via `useScroll` + `useTransform`:
- `rotateY`: interpolate `[0, 0.5, 1]` → `[-2, 0, 2]` degrees
- `rotateX`: interpolate `[0, 0.5, 1]` → `[1, 0, -1]` degrees
- `scale`: interpolate `[0, 0.5, 1]` → `[0.98, 1.02, 0.98]`
- `translateY`: interpolate `[0, 0.5, 1]` → `[-8, 0, 8]` px

All transforms wrapped in `style={{ perspective: "1200px" }}`.

Includes a gold ambient glow behind the phone (radial gradient, pulsing opacity via `glow-pulse` animation).

Accepts `children` (the PhoneFrame + PhoneScreenManager).

Hidden on mobile (`hidden lg:block`).

```tsx
"use client";

import { useRef } from "react";
import { motion, useScroll, useTransform } from "motion/react";

interface StickyPhoneProps {
  children: React.ReactNode;
}

export function StickyPhone({ children }: StickyPhoneProps) {
  const containerRef = useRef<HTMLDivElement>(null);
  const { scrollYProgress } = useScroll();

  const rotateY = useTransform(scrollYProgress, [0, 0.5, 1], [-2, 0, 2]);
  const rotateX = useTransform(scrollYProgress, [0, 0.5, 1], [1, 0, -1]);
  const scale = useTransform(scrollYProgress, [0, 0.5, 1], [0.98, 1.02, 0.98]);
  const y = useTransform(scrollYProgress, [0, 0.5, 1], [-8, 0, 8]);

  return (
    <div
      ref={containerRef}
      className="hidden lg:flex items-start justify-center"
    >
      <div className="sticky top-[50vh] -translate-y-1/2" style={{ perspective: "1200px" }}>
        {/* Ambient glow */}
        <div
          className="pointer-events-none absolute top-1/2 left-1/2 -z-10 h-[500px] w-[500px] -translate-x-1/2 -translate-y-1/2 rounded-full"
```
